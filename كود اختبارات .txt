    let currentQuestion = 0;
    let score = 0;
    let timer;
    let timeLeft = 30;
    let currentLanguage = "en";

    function loadQuestion() {
      const q = questions[currentQuestion];
      document.getElementById("question").innerText = currentLanguage === "en" ? q.question_en : q.question_ar;
      const optionsDiv = document.getElementById("options");
      optionsDiv.innerHTML = "";
      const options = currentLanguage === "en" ? q.options_en : q.options_ar;
      options.forEach(option => {
        const btn = document.createElement("button");
        btn.innerText = option;
        btn.className = "option-btn";
        btn.onclick = () => checkAnswer(option);
        optionsDiv.appendChild(btn);
      });
      document.getElementById("result").innerText = "";
      resetTimer();
    }

    function checkAnswer(answer) {
      if(answer.includes(questions[currentQuestion].answer)) {
        score++;
        document.getElementById("result").innerText = currentLanguage === "en" ? "Correct Answer" : "إجابة صحيحة";
        document.getElementById("result").style.color = "green";
      } else {
        document.getElementById("result").innerText = currentLanguage === "en" ? "Wrong Answer" : "إجابة خاطئة";
        document.getElementById("result").style.color = "red";
      }
      clearInterval(timer);
    }

    function nextQuestion() {
      currentQuestion++;
      if(currentQuestion < questions.length) {
        loadQuestion();
      } else {
        document.getElementById("question").innerText = currentLanguage === "en" ? "كفو بالتوفيق يارب !" : "انتهى الاختبار";
        document.getElementById("options").innerHTML = "";
        document.getElementById("result").innerText = (currentLanguage === "en" ? "Your score: " : "نتيجتك: ") + score + " / " + questions.length;
        document.getElementById("timer").innerText = "";
      }
    }

    function prevQuestion() {
      if(currentQuestion > 0) {
        currentQuestion--;
        loadQuestion();
      }
    }

    function toggleLanguage() {
      currentLanguage = currentLanguage === "en" ? "ar" : "en";
      document.getElementById("translate-btn").innerText = currentLanguage === "en" ? "Translate to Arabic" : "ترجمة إلى الإنجليزية";
      loadQuestion();
    }

    function resetTimer() {
      clearInterval(timer);
      timeLeft = 30;
      document.getElementById("timer").innerText = currentLanguage === "en" ? "Time: " + timeLeft + " seconds" : "الوقت: " + timeLeft + " ثانية";
      timer = setInterval(() => {
        timeLeft--;
        document.getElementById("timer").innerText = currentLanguage === "en" ? "Time: " + timeLeft + " seconds" : "الوقت: " + timeLeft + " ثانية";
        if(timeLeft <= 0) {
          clearInterval(timer);
          document.getElementById("result").innerText = currentLanguage === "en" ? "Time's up!" : "انتهى الوقت!";
          nextQuestion();
        }
      }, 1000);
    }

    loadQuestion();
